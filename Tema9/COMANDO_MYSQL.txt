DROP DATABASE concesionario;
CREATE DATABASE concesionario;

create table cliente (
	cliente_id INT PRIMARY KEY AUTO_INCREMENT, 
	documento_identidad CHAR(9) UNIQUE, 
	nombre VARCHAR(100) NOT NULL, 
	apellidos VARCHAR(100) NOT NULL, 
	email VARCHAR(20), telefono INT, 
	direccion VARCHAR(20), 
	fecha_nacimiento VARCHAR(15), 
	fotografia VARCHAR(10));

CREATE TABLE VEHICULO(
	MATRICULA CHAR(7) UNIQUE PRIMARY KEY, 
	MARCA VARCHAR(10) NOT NULL, 
	MODELO VARCHAR(10), 
	POTENCIA INT, 
	VELOCIDAD_MAX INT,
	IMAGEN VARCHAR(10) NOT NULL);

CREATE TABLE ALQUILER(
	ALQUILER_ID INT(100) PRIMARY KEY AUTO_INCREMENT, 
	CLIENTE_ID INT(100) NOT NULL,
	MATRICULA CHAR(7) NOT NULL, 
	LUGAR_RECOGIDA VARCHAR(50),
	FECHA_RECOGIDA VARCHAR(10),
	FECHA_DEVOLUCION VARCHAR(10),
	PRECIO DECIMAL(3,2),
	FOREIGN KEY(CLIENTE_ID) REFERENCES CLIENTE(CLIENTE_ID),
	FOREIGN KEY(MATRICULA) REFERENCES VEHICULO(MATRICULA));

CREATE TABLE USUARIO(
	ID INT(100) PRIMARY KEY AUTO_INCREMENT, 
	NOMBRE VARCHAR(10) NOT NULL, 
	APELLIDOS VARCHAR(10), 
	FECHA_DE_NACIMIENTO VARCHAR(10), 
	LOGIN VARCHAR(20) UNIQUE,
	PASSWORD VARCHAR(100),
	GRUPO VARCHAR(10));

CREATE TABLE IDIOMAS(
	ESPAÑOL VARCHAR(20),
	INGLES VARCHAR(20),
	CATALAN VARCHAR(20),
	EUSKERA VARCHAR(20))
;
